安装

pip install pypinyin

使用

from pypinyin import lazy_pinyin,Style
word = "中国"
for s in Style:
    pinyin = lazy_pinyin(word,style=s)
    print(f"{s.name:<15} = {s:<2} {" ".join(pinyin)}")
