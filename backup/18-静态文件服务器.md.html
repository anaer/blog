path-prefix: dufs

  - '@/:ro'  
  - 'user123:pass234@/:rw'

allow-delete: true  
allow-search: true  
allow-symlink: false  
allow-archive: false  
enable-cors: false

miniserve

```sh start.sh 启动服务

--mkdir 允许创建目录

--upload-files 需要指定路径, 该路径相对于根目录

--auth-file 授权文件

--qrcode 显示二维码

nohup /root/miniserve/miniserve --auth-file auth.txt --mkdir --upload-files / /root/miniserve/web &

```sh  
# stop.sh 停止服务  
pkill -f "miniserve"

user=$1  
pass=$2  
pw=$(echo -n ${pass} | sha256sum | cut -f 1 -d ' ')  
echo ${user}:sha256:$pw >> auth.txt  

# 添加用户
